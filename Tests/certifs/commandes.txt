
### Générer une paire de clés X25519 avec WireGuard puis empaqueter dans un certificat X509 (marche pas car X25519 pas supporté par openssl/X509) ###

wg genkey | tee privatekey | wg pubkey > publickey

openssl req -new -x509 -key privatekey -out cert.pem

openssl x509 -in cert.pem -pubkey -noout > public_key.pem

### Générer une paire de clés X25519 avec openssl au lieu de wireguard (marche pas pour l'étape 2) ###

openssl genpkey -algorithm X25519 -outform der | tail -c 32 | base64 > key.pem
openssl pkey -pubout -inform der -outform der -in key.pem -out pubkey.pem
